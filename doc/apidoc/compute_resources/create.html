POST /api/compute_resources
Create a compute resource.

Examples

POST /api/compute_resources
{
  "compute_resource": {
    "provider": "EC2",
    "name": "special_compute",
    "url": "eu-west1"
  }
}
200
{
  "compute_resource": {
    "provider": "EC2",
    "user": null,
    "created_at": "2012-11-11T08:07:52Z",
    "updated_at": "2012-11-11T08:07:52Z",
    "name": "special_compute",
    "id": 4,
    "url": "eu-west1",
    "description": null
  }
}
